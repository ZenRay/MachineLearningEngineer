def performance_metric2(y_true, y_predict):
    """"""
    ss_res = sum([(true - predict) ** 2 for true, predict in zip(y_true, y_predict)])
    ss_tot = sum([(true - sum(y_true) / len(y_true)) ** 2 for true in y_true])
    score = 1 - ss_res / ss_tot * 1.0

    return score

from sklearn.model_selection import KFold
from sklearn.metrics import make_scorer
from sklearn.model_selection import GridSearchCV
from sklearn.tree import DecisionTreeRegressor

def fit_model(X, y):
    """  [X,y]"""
    
    cross_validator = KFold(n_splits=3, shuffle=True, random_state=42)
    
    regressor = DecisionTreeRegressor()

    params = {"max_depth": np.arange(1, 11)}

    scoring_fnc = make_scorer(performance_metric)

    grid = GridSearchCV(estimator=regressor, param_grid=params, scoring=scoring_fnc, cv=cross_validator)

    #  [X,y]
    grid = grid.fit(X, y)

    # 
    return grid.best_estimator_
