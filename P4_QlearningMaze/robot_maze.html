import random

actions = ['u','r','d','l']
qline = {'u':1.2, 'r':-2.1, 'd':-24.5, 'l':27}
epsilon = 0.3 # 0.3

def choose_action(epsilon):
    
    action = None
    if epsilon == 0.3: # 
        action = random.choice(actions) # 
    else: 
        action = sorted(qline, key=lambda x:qline[x], reverse=True)[0] #  Q 
    return action
